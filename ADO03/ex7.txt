using System.Runtime.InteropServices.Marshalling;
using System.Security;
using System.Xml.Serialization;

namespace exerc_ciosC_;

class Program
{
    static void Main(string[] args)
    {
        Console.WriteLine("===================");
        Console.WriteLine("==PROGRAMA DO IMC==");
        Console.WriteLine("===================");
        Console.WriteLine();

        Console.Write("Me informe a sua altura: ");
        double altura = Convert.ToDouble(Console.ReadLine());
        Console.WriteLine();

        Console.Write("Me informe o seu peso: ");
        double peso = Convert.ToDouble(Console.ReadLine());
        Console.WriteLine();

        double imc = CalcularImc(altura, peso);
        bool situacao = VerificarIMCIdeal(imc);

        Console.WriteLine($"Seu IMC é: {imc}");
        Console.WriteLine($"É o IMC ideal? {situacao}");


    }
    public static double CalcularImc(double altura, double peso)
    {
        double imc = peso / (altura * altura);

        return imc;
    }


    public static bool VerificarIMCIdeal(double imc)
    {
        if (imc >= 18.5 && imc <= 24.9)
        {
            return true;
        }

        else
        {
            return false;
        }
    }



}
